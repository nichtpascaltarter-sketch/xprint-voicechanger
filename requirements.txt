# X-Print Voice Changer - Dependencies
# =====================================================================
# Optimiert für Python 3.15.0a8 (Alpha) - ohne harte Versions-Pins,
# damit pip die neuesten 3.15-kompatiblen Wheels findet.
#
# Installieren mit:
#   pip install --pre -r requirements.txt
#
# Falls einzelne Wheels fehlen (typisch in der Alpha-Phase):
#   pip install --pre --no-binary=:all: <paketname>
# (kompiliert aus dem Sourcecode - benötigt MSVC Build Tools)
# =====================================================================

# --- GUI ---
# Pure-Python (Tkinter-Wrapper). Funktioniert mit jeder 3.x-Version.
customtkinter>=5.2.2

# --- Audio I/O ---
# Pure-Python-Wrapper um PortAudio. CFFI-basiert, lädt fertige DLL mit.
sounddevice>=0.5.1

# --- Numerik (C-Extension) ---
# Numpy 2.1+ liefert moderne Wheels und arbeitet mit cpython-stable-ABI.
numpy>=2.1.0

# --- Audio-Effekte (C++-Extension!) ---
# pedalboard ist die kritischste Dependency: C++ via pybind11.
# Ab Version 0.9.20 mit breiterem Python-Version-Support.
# Falls für 3.15 noch kein Wheel existiert -> siehe README "Alpha-Build".
pedalboard>=0.9.20
